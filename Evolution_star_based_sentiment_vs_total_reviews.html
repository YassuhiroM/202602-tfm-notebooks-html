# 1) Load raw data (important: keep createTime + text columns)
df_raw = pd.read_csv("Reviews_EN.csv")

# 2) Parse date + year_month
df_raw["date"] = pd.to_datetime(df_raw["createTime"], errors="coerce", utc=True).dt.tz_convert(None)
df_raw = df_raw.dropna(subset=["date"])  # if any rows have invalid createTime
df_raw["year_month"] = df_raw["date"].dt.to_period("M")

# 3) Rating numeric (same mapping as your notebook)
rating_map = {"ONE": 1, "TWO": 2, "THREE": 3, "FOUR": 4, "FIVE": 5}
df_raw["rating"] = df_raw["starRating"].map(rating_map)

# 4) Define "has text" (use review_clean if it exists; fallback to comment)
text_col = "review_clean" if "review_clean" in df_raw.columns else "comment"
df_raw[text_col] = df_raw[text_col].fillna("").astype(str)
df_raw["has_text"] = df_raw[text_col].str.strip().ne("")

# 5) Sentiment from rating (3-way)
def rating_to_sentiment(r):
    if pd.isna(r):
        return np.nan
    if r >= 4:
        return "Positive"
    if r == 3:
        return "Neutral"
    return "Negative"

df_raw["sentiment_3way"] = df_raw["rating"].apply(rating_to_sentiment)

# 6) Filter range (>= 2022-01)
df_plot = df_raw[df_raw["year_month"] >= pd.Period("2022-01")].copy()

# --- Build stacked % bars (sentiment mix) ---
sent_counts = (
    df_plot.dropna(subset=["sentiment_3way"])
          .groupby(["year_month", "sentiment_3way"])
          .size()
          .unstack(fill_value=0)
)

# Ensure consistent column order
for c in ["Positive", "Neutral", "Negative"]:
    if c not in sent_counts.columns:
        sent_counts[c] = 0
sent_counts = sent_counts[["Positive", "Neutral", "Negative"]]

sent_totals = sent_counts.sum(axis=1)
sent_perc = sent_counts.div(sent_totals, axis=0) * 100


# --- Build a single table with counts + % + totals ---
table = pd.DataFrame(index=sent_counts.index.astype(str))

# Totals (two definitions)
table["total_reviews_all_rows"] = total_all.values
table["total_reviews_non_empty_text"] = total_text.values
table["share_with_text_%"] = (table["total_reviews_non_empty_text"] / table["total_reviews_all_rows"] * 100).round(1)

# Sentiment counts + percentages (based on rating->sentiment)
for s in ["Positive", "Neutral", "Negative"]:
    table[f"{s.lower()}_n"] = sent_counts[s].values
    table[f"{s.lower()}_%"] = sent_perc[s].round(1).values

# Optional: nicer ordering of columns
table = table[
    [
        "total_reviews_all_rows",
        "total_reviews_non_empty_text",
        "share_with_text_%",
        "positive_n", "positive_%",
        "neutral_n", "neutral_%",
        "negative_n", "negative_%",
    ]
]

# Show table
table



# --- Build the two total-volume lines ---
total_all = df_plot.groupby("year_month").size().reindex(sent_perc.index, fill_value=0)
total_text = (
    df_plot[df_plot["has_text"]]
    .groupby("year_month")
    .size()
    .reindex(sent_perc.index, fill_value=0)
)

# 7) Plot (similar to your example)
fig, ax1 = plt.subplots(figsize=(18, 7))

colors = {"Positive": "#2ecc71", "Neutral": "#f1c40f", "Negative": "#e74c3c"}
sent_perc.plot(kind="bar", stacked=True, ax=ax1, color=[colors[c] for c in sent_perc.columns], width=0.85)

ax1.set_title("(Star based) sentiment vs. Total Reviews",
              fontsize=18,
              fontweight="bold",
              pad=14
              )
ax1.set_xlabel("Year-Month")
ax1.set_ylabel("Percentage of Reviews (%)")
ax1.set_ylim(0, 100)

# Second axis for totals
ax2 = ax1.twinx()
x = np.arange(len(sent_perc.index))

ax2.plot(x, total_all.values, marker="o", linewidth=2, label="Total Reviews (all rows)")
ax2.plot(x, total_text.values, marker="o", linewidth=2, linestyle="--", label="Total Reviews (non-empty text)")
ax2.set_ylabel("Total Number of Reviews")

# Make x labels readable
ax1.set_xticklabels([str(p) for p in sent_perc.index], rotation=45, ha="right")

# Legends
ax1.legend(loc="upper left", title="Sentiment")
ax2.legend(loc="upper right")

plt.tight_layout()
plt.show()
